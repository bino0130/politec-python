## 출력 5-마지막 문자의 예제 2
a="아침" ## a = 아침
b="점심" ## b = 점심
c="저녁" ## c = 저녁
print(a,b,c) ## a,b,c 출력
print(a,b,c, sep = "먹고 ") ## a b c 사이에 먹고 출력
print(a,b,c, sep ="먹고 ", end = "") ## 문장 출력 후 뒤에 문장이 해당 문장 뒤에 붙어서 출력
print(a,b,c, sep ="자고 ", end = "") ## 문장 출력 후 뒤에 문장이 해당 문장 뒤에 붙어서 출력
print(a,b,c, sep="먹고 ", end = "\n") ## 문장 출력 후 문장 끝에서 개행
print(a,b,c, sep="자고 ", end = "\n") ## 문장 출력 후 문장 끝에서 개행
